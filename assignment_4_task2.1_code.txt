		diff=[]
		tmp=0
		for index,i in df.iterrows():    
			if i['X']==0:
				tmp=index
				diff.append(i['X'])
			else:
				if index>=tmp:
					diff.append(index-tmp)
				else:
					diff.append(index)
		print(diff)  #print difference
		df['out']=pd.Series(diff,dtype=int)
